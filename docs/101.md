## **216. 三七二一**

```c++
#include<iostream>
using namespace std;
typedef long long ll;
ll count(ll x,ll k){        //求出x每一位中共有几个k 
	ll cnt=0;				//计数器cnt表示当前找到几个k 
	do{						//重复循环	
		if(x%10==k)cnt++;	//如果x末尾是k，计数器cnt加1 
		x/=10;				//删除x末尾数字 
	}while(x);				//当x非零时，重复循环 
	return cnt;
}
int main(){
	ll n;					//定义长整数变量n 
	cin>>n;					//输入n	 
	if(count(n,3)&&count(n,7)&&count(n,2)&&count(n,1))  //如果n里有3并且n里有7并且n里有2并且n里有1 
		cout<<"Yes"<<endl;	//输出Yes 
	else 					//否则 
		cout<<"No"<<endl;	//输出No 
	return 0;
}
```

## **797. 幸运整数**

```c++
#include<iostream>
using namespace std;
typedef long long ll;
bool lucky(ll x){
	do{
		ll r=x%10;			//取出x末尾数字赋值给r 
		if(r!=5 && r!=8)	//如果该位数字不是5并且也不是8 
			return 0;		//返回结果，x不是幸运数字 
		x/=10;				//删除x末尾数 
	}while(x);				//当x非零时重复循环 
	return 1;				//返回结果，x是幸运数字
}
int main() {
	ll k,n=5,i=1;			//i代表当前有几个幸运整数 
	cin>>k;
	while(i<k){				//当目前幸运整数不够 
		n++;				//答案n加一 
		if(lucky(n))
			i++;			//如果n是幸运整数，计数器加一 
	}
	cout<<n<<endl;			//输出n，换行 
	return 0;
}
```

## 306.盗墓2

```cpp
/*
g[i]代表拿第i件宝物时在前i件里最大拿多少价值 
*/
#include<bits/stdc++.h>
using namespace std;
const int N=109;
int n,g[N],x[N];
int main(){
	freopen("tomb.in","r",stdin);
	freopen("tomb.out","w",stdout);
	cin>>n;
	for(int i=1;i<=n;i++) cin>>x[i];
	g[0]=0; g[1]=x[1]; g[2]=x[2]; g[3]=x[3];
	for(int i=4;i<=n;i++){
		int big=g[0];
		for(int j=1;j<=i-3;j++) 
			big=max(big,g[j]);
		g[i]=big+x[i];
	}
	int ans=g[0];
	for(int i=1;i<=n;i++)
		ans=max(ans,g[i]);
	cout<<ans<<endl;
	return 0;
}
```
