## 2. 删除最大和最小

```c++
#include<iostream>
using namespace std;
int a,f[10000];
int main ()
{
  cin>>a;
  for (int i=0; i<a; i++)
    cin>>f[i];
  int large=f[0];
  int small=f[0];
  for (int i=1; i<a; i++)
  {
	if(f[i]>large)
		large=f[i];
    if(f[i]<small)
        small=f[i];
  }
  for (int i=0; i<a; i++)
    if(f[i]!=large&&f[i]!=small)
      cout<<f[i]<<" ";
  return 0;
}

```
